1 /**
  2  * @fileOverview This is a lightweight utility for javascript inheritance
  3  * heavily inspired by the Closure library {@code goog.inherits} and
  4  * {@code goog.base}.  Note: This is NOT compatible with IE8 (and it probably
  5  * never will be) due to {@link Object.create} and
  6  * {@link Object.getOwnPropertyDescriptor}.
  7  */
  8 /**
  9  * @namespace
 10  * @name grail
 11  */
 12 goog.provide('grail');
 13 
 14 
 15 /**
 16  * @const {string}
 17  * @private
 18  */
 19 grail.SUPER_PROP_ = '_grail_super_';
 20 
 21 
 22 /**
 23  * Extends the prototype chain of {@link childCtor} from
 24  * {@link parentCtor.prototype}.
 25  * @param {!Function} childCtor The child class constructor.
 26  * @param {!Function} parentCtor The parent class constructor.
 27  * @return {!Function} The augmented {@link childCtor}.
 28  */
 29 grail.inherit = function(childCtor, parentCtor) {
 30   if (childCtor == parentCtor ||
 31       childCtor.prototype.isPrototypeOf(parentCtor.prototype)) {
 32     throw Error('Invalid circular inheritance');
 33   } else if (childCtor.prototype[grail.SUPER_PROP_]) {
 34     throw Error('Invalid multiple inheritance');
 35   }
 36 
 37   var preservedProperties =
 38       grail.getOwnPropertyDescriptors_(childCtor.prototype);
 39   childCtor.prototype =
 40       Object.create(parentCtor.prototype, preservedProperties);
 41 
 42   /** @override */
 43   childCtor.prototype.constructor = childCtor;
 44 
 45   Object.defineProperty(childCtor.prototype, grail.SUPER_PROP_, {
 46     'configurable': false,
 47     'enumerable': false,
 48     'value': parentCtor.prototype,
 49     'writable': false
 50   });
 51   return childCtor;
 52 };
 53 
 54 
 55 /**
 56  * Either calls the parent constructor if this is the instantiation of
 57  * {@link scope} or calls the super method of {@link scope}. The method or super
 58  * constructor will be called with all arguments minus the first 2.
 59  * @param {!Object} scope
 60  * @param {string|!Function} methodNameOrCtor If {@code methodNameOrCtor} is the
 61  *     {@code constructor} of {@code scope} then the super constructor will be
 62  *     called. Otherwise the super method named {@code methodNameOrCtor} will be
 63  *     called.
 64  * @param {...*} var_args The arguments to pass to the super method.
 65  * @return {*} The result of the super method
 66  */
 67 grail.call = function(scope, methodNameOrCtor, var_args) {
 68   var superProto = scope[grail.SUPER_PROP_];
 69   if (!superProto) {
 70     throw Error('Cannot find superclass');
 71   }
 72 
 73   var args = Array.prototype.slice.call(arguments, 2);
 74 
 75   var isInstantiation = scope.constructor == methodNameOrCtor;
 76   if (isInstantiation) {
 77     return superProto.constructor.apply(scope, args);
 78   } else {
 79     var method = superProto[methodNameOrCtor];
 80     if (!method) {
 81       throw Error('Method not found ' + methodNameOrCtor);
 82     }
 83     return method.apply(scope, args);
 84   }
 85 };

 88 /**
 89  * Returns an object of property descriptors for all properties directly in
 90  * {@link propertyObject}.
 91  * @param {!Object} propertyObject
 92  * @return {!Object}
 93  * @private
 94  */
 95 grail.getOwnPropertyDescriptors_ = function(propertyObject) {
 96   var descriptors = {};
 97   var properties = Object.getOwnPropertyNames(propertyObject);
 98   for (var i = 0; i < properties.length; i++) {
 99     var prop = properties[i];
100     descriptors[prop] = Object.getOwnPropertyDescriptor(propertyObject, prop);
101   }
102   return descriptors;
103 };

106 /**
107  * Exports the API to the global namespace in order to avoid removal during
108  * Closure compilation.
109  */
110 (function(scope) {
111   scope['grail'] = grail;
112   scope['grail']['inherit'] = grail.inherit;
113   scope['grail']['call'] = grail.call;
114 })(this);
